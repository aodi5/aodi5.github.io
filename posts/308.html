Chaotic maps

1. Chebyshev map

$$
x_{k+1}=cos(kcos^{-1}(x_k))
$$

2. Circle map

$$
x_{k+1}=x_k+b-(\frac{a}{2\pi})sin(2\pi x_k))mod(1)\
a=0.5 \
b=0.2
$$

3. Gauss/mouse map

$$
x_{k+1} = \begin{cases} 0 & x_k = 0 \
\frac{1}{x_kmod(1)} & otherwise
\end{cases}\
\frac{1}{x_kmod(1)}=\frac{1}{x_k}-[\frac{1}{x_k}]
$$

4. Intermittency map

$$
x_{k+1} = \begin{cases}
\varepsilon+x_k+cx_k^n & 0<x_k \leq P \
\frac{x_k-P}{1-P} & P<x_k<1
\end{cases}\
c=\frac{1 - \varepsilon -P}{P^n},\ n=2
$$

5. Iterative map

$$
x_k+1=sin(\frac{a\pi}{x_k})\a\in(0,1)
$$

6. Liebovitch map

$$
x_{k+1} = \begin{cases}
\alpha x_k & 0<x_k \leq P_1 \
\frac{P-x_k}{P_2-P_1} & P_1<x_k\leq P_2\1-\beta(1-x_k) &P_2<x_k\leq1
\end{cases}\
\alpha<\beta,并且\
\alpha=\frac{P_2}{P_1}(1-(P_2-P_1))\
\beta=\frac{1}{P_2-1}((P_2-1)-P_1(P_2-P_1))
$$

7. Logistic map

$$
x_{k+1}=ax_k(1-x_k)\
x_k是第k个混沌数，k为迭代次数，
x\in(0,1),x_0\notin(0,0.25,0.5,0.75,1),a为实数(a=4)
$$

8. Piecewise map

$$
x_{k+1} = \begin{cases}
\frac{x_k}{P} & 0\leq x_k < P \
\frac{x_k-P}{0.5-P} & P\leq x_k<\frac{1}{2}\
\frac{1-P-x_k}{0.5-P} & \frac{1}{2}\leq x_k < 1-P \
\frac{1-x_k}{P} & 1-P \leq x_k <1
\end{cases} \
P \in (0,1)为控制参数，x \in (0,1)
$$

9. Sine map

$$
x_{k+1}=\frac{a}{4}sin(\pi x_k), \a\in (0,4]
$$

10. Singer map

$$
x_{k+1}= \mu (7.86x_k- 23.31 x_k^2 + 28.75 x_k^3- 13.302875 x_k^4)\
\mu \in (0.9,1.08)
$$

11. Sinusoidal map

$$
x_{k+1}=a x_k^2 sin(\pi x_k)\
a=2.3,x_0=0.7 时有简化形式：\
x_{k+1} = sin(\pi x_k)
$$

12. Tent map

The tent map is very similar to the logistic map.
$$
x_{k+1} = \begin{cases}
\frac{x_k}{0.7} &x_k<0.7 \
\frac{10}{3}(1 - x_k) & x_k \geq 0.7
\end{cases}
$$

$$
x_{k+1} = \begin{cases}
\frac{x_k}{\alpha} &x_k<\alpha \
\frac{(1 - x_k)}{1-\alpha} & x_k \geq \alpha
\end{cases}
\alpha \in (0,1)
$$

13.β-chaotic

$$
x_{t+1}=k\beta(x_t ; \mu ; v ; x_1 ; x_2)\
\beta(x_t ; \mu ; v ; x_1 ; x_2)=\begin{cases}
(\frac{x-x_1}{x_c - x_1})^ \mu (\frac{x_2 - x}{x_2 - x_c})^v & if \ x \in[x_1,x_2] \
0 & otherwise
\end{cases}\
\beta(x_t ; \mu ; v ; x_1 ; x_2) \in R, x_1<x_2.\
x_c=\frac{\mu x_2 + v x_1}{\mu + v}\
\mu = b_1 + c_1 \times a,v=b_2 + c_2 \times a\
b_i,c_i为实数
例： a=[−0.8:0.7], x =− 10 , x = 1 1 , k=0.85, b = 5 1 , c = 1 1 , b = 3 2 and c =− 12
$$

14. Cubic map

$$
x_{k+1}= \rho(1-x_k^2)
$$

import matplotlib.pyplot as plt
import numpy as np
import math


def Chebyshevmap1(x0, max_g):
    """
    Chebyshev 映射
    :param x0: 初值
    :param max_g:最大迭代次数
    :return: x的列表
    """
    x = x0
    x_list = []
    for i in max_g:
        x = math.cos(i * math.cos(x) ** (-1))
        x_list.append(x)
    return x_list


def draw1(x0, iters):
    """
    显示Chebyshev 映射图像
    :param x0: 初值
    :param iters: 迭代次数
    :return: 图像
    """
    max_g = np.linspace(1, iters, num=iters)
    x_list = Chebyshevmap1(x0, max_g)
    plt.plot(max_g, x_list)
    plt.title('Chebyshev map')
    plt.xlabel('iters')
    plt.ylabel('x')
    plt.xlim(0, 500)
    plt.ylim(0, 1)
    plt.show()


draw1(0.7, 500)


def Circlemap2(a, b, x0, max_g):
    a = a
    b = b
    x = x0
    x_list = []
    for i in max_g:
        x = (x + b - ((a / (2 * math.pi)) * math.sin(2 * math.pi * x))) % 1
        x_list.append(x)
    return x_list


def draw2(a, b, x0, iters):
    max_g = np.linspace(1, iters, num=iters)
    x_list = Circlemap2(a, b, x0, max_g)
    plt.plot(max_g, x_list)
    plt.title('Circle map')
    plt.xlabel('iters')
    plt.ylabel('x')
    plt.xlim(0, 500)
    plt.ylim(0, 1)
    plt.show()


draw2(0.5, 0.2, 0.7, 500)


def Gaussmap3(x0, max_g):
    x = x0
    x_list = []
    for i in max_g:
        if x == 0:
            x = 0
        else:
            x = 1 / x - int(1/x)
            x_list.append(x)
    return x_list


def draw3(x0, iters):

draw3(0.7, 500)


def Intermittency4(epsilon, x0, P, max_g):
    x = x0
    x_list = []
    for i in max_g:
        if 0 < x <= P:
            x = epsilon + x + ((1 - epsilon - P) / (P ** 2)) * (x ** 2)
        
        else:
            x = (x - P) / (1 - P)
        x_list.append(x)
    return x_list


def draw4(epsilon, x0, P, iters):
    max_g = np.linspace(1, iters, num=iters)
    x_list = Intermittency4(epsilon, x0, P, max_g)
    plt.plot(max_g, x_list)
    plt.title('Intermittency map')
    plt.xlabel('iters')
    plt.ylabel('x')
    plt.xlim(0, 500)
    plt.ylim(0, 1)
    plt.show()


draw4(0.07, 0.7, 0.5, 500)

def Liebovitchmap6(x0, P1, P2, P, max_g):
    x = x0
    x_list = []
    for i in max_g:
        if 0 < x <= P1:
            x = (P2 / P1) * (1 - (P2 - P1)) * x
        elif P1 < x <= P2:
            x = (P - x) / (P2 - P1)
        elif P2 < x <= 1:
            x = 1 - ((1 / (P2 - 1)) * ((P2 - 1) - P1 * (P2 - P1)) * (1 - x))
        x_list.append(x)
    return x_list


def draw6(x0, P1, P2, P, iters):
    max_g = np.linspace(1, iters, num=iters)
    x_list = Liebovitchmap6(x0, P1, P2, P, max_g)
    plt.plot(max_g, x_list)
    plt.title('Liebovitch map')
    plt.xlabel('iters')
    plt.ylabel('x')
    plt.xlim(0, 500)
    plt.ylim(0, 1)
    plt.show()


draw6(0.7, 0.1, 0.9, 0.9, 500)


def logisticmap7(u, x0, max_g):
    """
    logiatic map函数
    :param u: 在4附近混沌
    :param x0: 初值
    :param max_g: 迭代次数
    :return: 函数值
    """
    x = x0
    x_list = []
    for i in max_g:
        x = u * x * (1 - x)
        x_list.append(x)
    return x_list

def piecewisemap8(x0, P, max_g):
    x = x0
    x_list = []
    for i in max_g:
        if 0 <= x < P:
            x = x / P
        elif P <= x < 0.5:
            x = (x - P) / (0.5 - P)
        elif 0.5 <= x < 1 - P:
            x = (1 - P - x) / (0.5 - P)
        elif 1 - P <= x < 1:
            x = (1 - x) / P
        x_list.append(x)
    return x_list


def draw8(x0, P, iters):
    max_g = np.linspace(1, iters, num=iters)
    x_list = piecewisemap8(x0, P, max_g)

def tentmap12(alpha, x0, max_g):
    """
    tent map 函数
    :param alpha:0到1之间的参数
    :param x0: 初值
    :param max_g: 迭代次数
    :return: 函数值
    """
    x = x0
    x_list = []
    for i in max_g:
        if x < alpha:
            x = x / alpha
        else:
            x = (1 - x) / (1 - alpha)
        x_list.append(x)
    return x_list


def draw12(alpha, x0, iters):
    """
    画出tent map 的图像
    :param alpha:0到1之间的参数

    :return:图像

    max_g = np.linspace(1, iters, num=iters)

    plt.title('tent map')

def betamap13(x0, a, b1, b2, c1, c2, k, x1, x2, max_g):
    x = x0
    x_list = []
    u = b1 + c1 * a
    v = b2 + c2 * a
    xc = (u * x2 + v * x1) / (u + v)
    for i in max_g:
        if x1 <= x <= x2:
            x = k * (((x - x1) / (xc - x1)) ** u) * \
                (((x2 - x) / (x2 - xc)) ** v)
        else:
            x = 0
        x_list.append(x)
    return x_list


def draw13(x0, a, b1, b2, c1, c2, k, x1, x2, iters):
    max_g = np.linspace(1, iters, num=iters)
    x_list = betamap13(x0, a, b1, b2, c1, c2, k, x1, x2, max_g)
    plt.plot(max_g, x_list)
    plt.title('beta map')
    plt.xlabel('iters')
    plt.ylabel('x')
    plt.xlim(0, 500)
    plt.ylim(0, 1)
    plt.show()


draw13(0.7, -0.3, 5, 3, 0.01, -0.01, 0.9, -1, 1, 500)
